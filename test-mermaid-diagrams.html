測試 4: ClassDiagram with Interface

classDiagram
    class Client {
        +request()
    }
    class Context {
        -strategy Strategy
        +setStrategy(Strategy)
        +doSomething()
    }
    class Strategy {
        <<interface>>
        +execute()
    }
    class ConcreteStrategyA {
        +execute()
    }
    class ConcreteStrategyB {
        +execute()
    }
    
    Client --> Context
    Context --> Strategy
    Strategy <|.. ConcreteStrategyA
    Strategy <|.. ConcreteStrategyB
